#2.过滤信息,设置cate
SELECT * FROM omero_forage.post WHERE cate = '洗衣常识' AND title NOT REGEXP '赔|爆炸|失火|冒充|追偿|寻衣|缩水';

UPDATE omero_forage.post SET cate = '洗衣常识' WHERE cate = '干洗知识' AND title REGEXP '什么是|如何|[?]';
UPDATE omero_forage.post SET cate = '技术培训' WHERE cate = '干洗知识' AND title REGEXP '注意|事项|方法|窍门|技巧|图解';
UPDATE omero_forage.post SET cate = '产品中心' WHERE cate = '干洗知识' AND title REGEXP '干洗机|干洗溶剂|冷洗精|蒸汽熨斗';
UPDATE omero_forage.post SET cate = '产品中心' WHERE cate = '干洗器材'
UPDATE omero_forage.post SET cate = '运营指导' WHERE cate = '干洗创业'
UPDATE omero_forage.post SET cate = '负面新闻' WHERE cate = '干洗新闻' AND title REGEXP '赔|爆炸|失火|冒充|追偿|寻衣|缩水';

#3.设置cate_id
UPDATE post JOIN omero_pro_dev.news_cates d ON d.name = post.cate SET cate_id = d.id;

#4.导入数据
